236  CONSTEXPR
237  DataType data_type(csv::string_view in, long double* const out) {
238  // Empty string --> NULL
239  if (in.size() == 0)
240  return CSV_NULL;
241 
242  bool ws_allowed = true,
243  neg_allowed = true,
244  dot_allowed = true,
245  digit_allowed = true,
246  has_digit = false,
247  prob_float = false;
248 
249  unsigned places_after_decimal = 0;
250  long double integral_part = 0,
251  decimal_part = 0;
252 
253  for (size_t i = 0, ilen = in.size(); i < ilen; i++) {
254  const char& current = in[i];
255 
256  switch (current) {
257  case ' ':
258  if (!ws_allowed) {
259  if (isdigit(in[i - 1])) {
260  digit_allowed = false;
261  ws_allowed = true;
262  }
263  else {
264  // Ex: '510 123 4567'
265  return CSV_STRING;
266  }
267  }
268  break;
269  case '-':
270  if (!neg_allowed) {
271  // Ex: '510-123-4567'
272  return CSV_STRING;
273  }
274 
275  neg_allowed = false;
276  break;
277  case '.':
278  if (!dot_allowed) {
279  return CSV_STRING;
280  }
281 
282  dot_allowed = false;
283  prob_float = true;
284  break;
285  case 'e':
286  case 'E':
287  // Process scientific notation
288  if (prob_float) {
289  size_t exponent_start_idx = i + 1;
290 
291  // Strip out plus sign
292  if (in[i + 1] == '+') {
293  exponent_start_idx++;
294  }
295 
296  return _process_potential_exponential(
297  in.substr(exponent_start_idx),
298  neg_allowed ? integral_part + decimal_part : -(integral_part + decimal_part),
299  out
300  );
301  }
302 
303  return CSV_STRING;
304  break;
305  default:
306  short digit = current - '0';
307  if (digit >= 0 && digit <= 9) {
308  // Process digit
309  has_digit = true;
310 
311  if (!digit_allowed)
312  return CSV_STRING;
313  else if (ws_allowed) // Ex: '510 456'
314  ws_allowed = false;
315 
316  // Build current number
317  if (prob_float)
318  decimal_part += digit / pow10(++places_after_decimal);
319  else
320  integral_part = (integral_part * 10) + digit;
321  }
322  else {
323  return CSV_STRING;
324  }
325  }
326  }
327 
328  // No non-numeric/non-whitespace characters found
329  if (has_digit) {
330  long double number = integral_part + decimal_part;
331  if (out) {
332  *out = neg_allowed ? number : -number;
333  }
334 
335  return prob_float ? CSV_DOUBLE : _determine_integral_type(number);
336  }
337 
338  // Just whitespace
339  return CSV_NULL;
340  }
341  }

279  CSV_INLINE bool CSVReader::read_row(CSVRow &row) {
280  while (true) {
281  if (this->records.empty()) {
282  if (this->records.is_waitable())
283  // Reading thread is currently active => wait for it to populate records
284  this->records.wait();
285  else if (this->eof())
286  // End of file and no more records
287  return false;
288  else {
289  // Reading thread is not active => start another one
290  if (this->read_csv_worker.joinable())
291  this->read_csv_worker.join();
292 
293  this->read_csv_worker = std::thread(&CSVReader::read_csv, this, internals::ITERATION_CHUNK_SIZE);
294  }
295  }
296  else if (this->records.front().size() != this->n_cols &&
297  this->_format.variable_column_policy != VariableColumnPolicy::KEEP) {
298  auto errored_row = this->records.pop_front();
299 
300  if (this->_format.variable_column_policy == VariableColumnPolicy::THROW) {
301  if (errored_row.size() < this->n_cols)
302  throw std::runtime_error("Line too short " + internals::format_row(errored_row));
303 
304  throw std::runtime_error("Line too long " + internals::format_row(errored_row));
305  }
306  }
307  else {
308  row = std::move(this->records.pop_front());
309  this->n_rows++;
310  return true;
311  }
312  }
313 
314  return false;
315  }
